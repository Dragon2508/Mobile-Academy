Создание первого приложения

Для демонстраци возможностей Xamarin, создадим простое приложение в котором будет находится круглая кнопка с номером, отображающим количество нажатий на неё. Для этого вставьте следующий код в файл MainPage.xaml:

<?xml version="1.0" encoding="utf-8" ?>
<ContentPage xmlns="http://xamarin.com/schemas/2014/forms"
xmlns:x="http://schemas.microsoft.com/winfx/2009/xaml"
x:Class="App3.MainPage">
<StackLayout>
<Button x:Name="MyButton" WidthRequest="80" HeightRequest="80" HorizontalOptions="Center" Text="0" BorderRadius="40" Clicked="Button_Clicked" />
</StackLayout>
</ContentPage>

При нажатии на кнопку, будет случайным образом меняться её местоположение и цвет.Для реализации данного приложения скопируйте ниже представленный код в MainPage.xaml.cs:

using System;
using Xamarin.Forms;
namespace App3
{
public partial class MainPage : ContentPage
{
public MainPage()
{
InitializeComponent();
}
Random rnd = new Random(); // Объявление экземпляра класса Random
int count = 0; // Объявление переменное для сохранения количества нажатий по кнопки
void Button_Clicked(object sender, System.EventArgs e) // Инициализация метода нажатия на кнопку
{
count++; // количество кликов +1
MyButton.TranslationX = rnd.Next(-150, 150); // Рандомное выставление кнопки по координате X
MyButton.TranslationY = rnd.Next(0, 550); // Рандомное выставление кнопки по координате Y
MyButton.Text = count.ToString(); // Задание текста кнопки
System.Drawing.Color color = System.Drawing.Color.FromArgb(rnd.Next(0, 255), rnd.Next(0, 255), rnd.Next(0, 255));
// Задание случайного цвета
MyButton.BackgroundColor = color; // Задание цвета кнопки
float lightness = color.GetBrightness(); // Определение яркости цвета
if (lightness >= 0.45)
// Если яркость больше или равна 0.45, то меняем цвет текста кнопки на чёрный, иначе на белый {
MyButton.TextColor = new Color(0, 0, 0);
}
else
{
MyButton.TextColor = new Color(1, 1, 1);
}
}
}
}
